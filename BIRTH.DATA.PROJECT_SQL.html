1    -- What is the sum of all births in Florida (2016-2018)?
2    SELECT sum(Births) as total_num_birth
3    FROM `bigquery-public-data.sdoh_cdc_wonder_natality.county_natality` 
4    where County_of_Residence like "%FL%"

8    --What is the sum of all birth in California in 2018?
9    SELECT sum(Births) as new_column
10   FROM `bigquery-public-data.sdoh_cdc_wonder_natality.county_natality` 
11   where County_of_Residence like "%CA%"
12   and Year = '2018-01-01'

16   --What county in NY has the highest Average Number of Prenatal Visits (descending order).
17   Select County_of_Residence, max(Ave_Number_of_Prenatal_Wks) AS highest_num_prenat
18   FROM `bigquery-public-data.sdoh_cdc_wonder_natality.county_natality`
19   WHERE County_of_Residence like "%NY%"
20   GROUP BY County_of_Residence
21   ORDER BY highest_num_prenat DESC

25   -- What county had the highest average age of mother in the overall table?
26   select County_of_Residence, max(Ave_Age_of_Mother) as highest_age_mom
27   from `bigquery-public-data.sdoh_cdc_wonder_natality.county_natality`
28   group by County_of_Residence
29   order by highest_age_mom desc

33   -- What is the count of all average birth weight in grams in the counties that has "GA"
34   select count(Ave_Birth_Weight_gms) as count_ave_birth_weight
35   from `bigquery-public-data.sdoh_cdc_wonder_natality.county_natality`
36   where County_of_Residence like "%GA%"

40   -- What county has the lowest Average number of prenatal weeks?
41   select County_of_Residence, min(Ave_Number_of_Prenatal_Wks) as lowest_prenat_wks
42   from `bigquery-public-data.sdoh_cdc_wonder_natality.county_natality`
43   group by County_of_Residence
44   order by lowest_prenat_wks

48   --- This gives us county of residence when we input the code.
49   SELECT County_of_Residence, County_of_Residence_FIPS
50       CASE  WHEN County_of_Residence_FIPS = 39049 then 'Franklin County, OH'
51            When County_of_Residence_FIPS = 13113 then 'Fayette County GA'
52            Else 'other' END
53            as new_colom
54   FROM `bigquery-public-data.sdoh_cdc_wonder_natality.county_natality`
55   order by new_colom
56

58   --- This gives us the first 10 counties with births lower than 1000 in all the states throughout the year 2016-2018.
59   select County_of_Residence, Births, Year
60   from `bigquery-public-data.sdoh_cdc_wonder_natality.county_natality`
61   where births < 1000
62   order by Births desc
63   limit 10

66   -- This helps us compare the same table with eachother using Self-ish Joins. 
67   -- This also helps us filter out rows with unidentified counties. 
68   Select b1.Ave_Age_of_Mother, b2.Ave_Age_of_Mother, b1.County_of_Residence as county_res
69   from `bigquery-public-data.sdoh_cdc_wonder_natality.county_natality` as b1
70   inner join `bigquery-public-data.sdoh_cdc_wonder_natality.county_natality` as b2
71   on b1.Ave_Age_of_Mother = b2.Ave_Age_of_Mother
72   where b1.County_of_Residence is not null 
73   order by b1.Ave_Age_of_Mother
